--Join

--natural join
select p.pid,p.name,d.did,d.vehicle
from passports p, driving_lic d
where p.pid=d.passport;

--left join
select p.pid,p.name,d.did,d.vehicle
from passports p, driving_lic d
where p.pid=d.passport(+);

--right join 
select p.pid,p.name,d.did,d.vehicle
from passports p, driving_lic d
where p.pid(+)=d.passport;

--full join 
select p.pid,p.name,d.did,d.vehicle
from passports p
full outer join driving_lic d
on p.pid=d.passport;
